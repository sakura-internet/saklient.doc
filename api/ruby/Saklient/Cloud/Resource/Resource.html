def api_deserialize(obj, wrapped = false)
  Saklient::Util::validate_type(wrapped, 'bool')
  root = nil
  record = nil
  rkey = _root_key
  if !(obj).nil?
    if !wrapped
      if !(rkey).nil?
        root = {}
        root[rkey.to_sym] = obj
      end
      record = obj
    else
      root = obj
      record = obj[rkey.to_sym]
    end
  end
  api_deserialize_impl(record)
  _on_after_api_deserialize(record, root)
end

def exists
  params = {}
  Saklient::Util::set_by_path(params, 'Filter.ID', [_id])
  Saklient::Util::set_by_path(params, 'Include', ['ID'])
  result = @_client.request('GET', _api_path, params)
  return result[:Count] == 1
end
