def connect_to_swytch(swytch)
  Saklient::Util::validate_type(swytch, 'Saklient::Cloud::Resources::Swytch')
  @_client.request('PUT', _api_path + '/' + Saklient::Util::url_encode(_id) + '/to/switch/' + Saklient::Util::url_encode(swytch._id))
  return reload
end
