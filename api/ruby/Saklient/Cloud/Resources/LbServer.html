def update_status(obj)
  connStr = obj[:ActiveConn]
  @_active_connections = 0
  @_active_connections = (connStr).to_s().to_i(10) if !(connStr).nil?
  status = obj[:Status]
  @_status = (status).nil? ? nil : status.downcase()
  return self
end
